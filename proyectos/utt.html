UTT Torneo: Sumo Autónomo

Evolución electrónica: Integración de PCB comercial y optimización de algoritmos de búsqueda.

01.

Contexto y Salto de Calidad

Para el torneo de robótica en la Universidad Tecnológica de Tijuana (UTT), el objetivo principal fue **optimizar la fiabilidad**. Se mantuvo la base mecánica (chasis) del proyecto RoboWar anterior, ya que demostró ser estructuralmente robusta, pero se rediseñó por completo el núcleo electrónico y la lógica algorítmica.


La Mejora Crítica: Pasamos de un sistema cableado a una PCB comercial. Esto redujo drásticamente el ruido eléctrico en los sensores analógicos, eliminó los fallos por desconexión debidos a impactos de alta inercia, y permitió una distribución de peso más limpia y un centro de gravedad más bajo.

04.

Algoritmos de Búsqueda y Persistencia

Se desarrolló una máquina de estados de 4 fases para la búsqueda de oponentes, acompañada de un sistema de persistencia. Esto permite que el robot no pierda a su objetivo si hay una lectura "ciega" por microsegundos durante giros rápidos.

state_machine.ino

// Máquina de estados para búsqueda activa
void buscarOponente() {
  unsigned long ahora = millis();
  unsigned long dt = ahora - tiempoBusqueda;

  // Cambia de fase cada 300ms
  if (dt > 300) {
    tiempoBusqueda = ahora;
    faseBusqueda = (faseBusqueda + 1) % 4;
  }

  // Patrón: Giro Der -> Avanza -> Giro Izq -> Avanza
  switch (faseBusqueda) {
    case 0: girarDerecha(velBusqueda); break;
    case 1: avanzar(velBusqueda); break;
    case 2: girarIzquierda(velBusqueda); break;
    case 3: avanzar(velBusqueda); break;
  }
}

void loop() {
  // ... (Lectura y filtrado de sensores) ...
  
  // Lógica de Persistencia
  if (ahora - tiempoUltimoDetectado < persistencia) {
    if (ultimoLado == 1) girarIzquierda(velGiro);
    else girarDerecha(velGiro);
  } else {
    // Si expira la persistencia, inicia máquina de estados
    buscarOponente(); 
  }
}

#include 

// Motores
#define AIN1  8
#define AIN2  7
#define PWMA  9
#define BIN1  3
#define BIN2  4
#define PWMB  5

// Sensores IR
#define DIST_LEFT   A7
#define DIST_RIGHT  A2
#define DIST_FRONT  A5

// Sensores de línea
QTRSensors qtr;
const uint8_t SensorCount = 2;
uint16_t sensorValues[SensorCount];
#define LINE_THRESHOLD 1000

// Parametros
int velAtaque       = 255;
int velGiro         = 180;
int velBusqueda     = 160;
int limiteOponente  = 400;
int persistencia    = 500;

// Variables
int distIzq, distDer, distFront;
int ultimoLado = 1;
unsigned long tiempoUltimoDetectado = 0;
unsigned long tiempoBusqueda = 0;
int faseBusqueda = 0;

void setup() {
  pinMode(AIN1, OUTPUT); pinMode(AIN2, OUTPUT);
  pinMode(BIN1, OUTPUT); pinMode(BIN2, OUTPUT);
  pinMode(PWMA, OUTPUT); pinMode(PWMB, OUTPUT);
  pinMode(6, OUTPUT); digitalWrite(6, HIGH);

  qtr.setTypeRC();
  qtr.setSensorPins((const uint8_t[]){10, 12}, SensorCount);
  Serial.begin(9600);
  detener();
  delay(5000);
}

void avanzar(int velocidad) {
  digitalWrite(AIN1, LOW); digitalWrite(AIN2, HIGH);
  digitalWrite(BIN1, LOW); digitalWrite(BIN2, HIGH);
  analogWrite(PWMA, velocidad); analogWrite(PWMB, velocidad);
}

void retroceder(int velocidad) {
  digitalWrite(AIN1, HIGH); digitalWrite(AIN2, LOW);
  digitalWrite(BIN1, HIGH); digitalWrite(BIN2, LOW);
  analogWrite(PWMA, velocidad); analogWrite(PWMB, velocidad);
}

void girarIzquierda(int velocidad) {
  digitalWrite(AIN1, HIGH); digitalWrite(AIN2, LOW);
  digitalWrite(BIN1, LOW); digitalWrite(BIN2, HIGH);
  analogWrite(PWMA, velocidad); analogWrite(PWMB, velocidad);
}

void girarDerecha(int velocidad) {
  digitalWrite(AIN1, LOW); digitalWrite(AIN2, HIGH);
  digitalWrite(BIN1, HIGH); digitalWrite(BIN2, LOW);
  analogWrite(PWMA, velocidad); analogWrite(PWMB, velocidad);
}

void detener() {
  analogWrite(PWMA, 0); analogWrite(PWMB, 0);
}

void buscarOponente() {
  unsigned long ahora = millis();
  unsigned long dt = ahora - tiempoBusqueda;
  if (dt > 300) {
    tiempoBusqueda = ahora;
    faseBusqueda = (faseBusqueda + 1) % 4;
  }
  switch (faseBusqueda) {
    case 0: girarDerecha(velBusqueda); break;
    case 1: avanzar(velBusqueda); break;
    case 2: girarIzquierda(velBusqueda); break;
    case 3: avanzar(velBusqueda); break;
  }
}

void loop() {
  qtr.read(sensorValues);
  uint16_t izq = sensorValues[0];
  uint16_t der = sensorValues[1];

  bool bordeIzq = izq > LINE_THRESHOLD;
  bool bordeDer = der > LINE_THRESHOLD;

  distIzq   = analogRead(DIST_LEFT);
  distDer   = analogRead(DIST_RIGHT);
  distFront = analogRead(DIST_FRONT);

  unsigned long ahora = millis();

  static int fIzq = 0, fDer = 0, fFront = 0;
  fIzq   = (fIzq * 3 + distIzq) / 4;
  fDer   = (fDer * 3 + distDer) / 4;
  fFront = (fFront * 3 + distFront) / 4;

  if (bordeIzq && !bordeDer) {
    retroceder(velAtaque); delay(200);
    girarDerecha(velAtaque); delay(200); return;
  }

  if (bordeDer && !bordeIzq) {
    retroceder(velAtaque); delay(200);
    girarIzquierda(velAtaque); delay(200); return;
  }

  if (bordeIzq && bordeDer) {
    retroceder(velAtaque); delay(350);
    girarDerecha(velAtaque); delay(200); return;
  }

  if (fFront > limiteOponente) {
    avanzar(velAtaque);
    tiempoUltimoDetectado = ahora; return;
  }

  if (fIzq > limiteOponente && fDer < limiteOponente - 30) {
    girarIzquierda(velGiro);
    ultimoLado = 1; tiempoUltimoDetectado = ahora; return;
  }

  if (fDer > limiteOponente && fIzq < limiteOponente - 30) {
    girarDerecha(velGiro);
    ultimoLado = -1; tiempoUltimoDetectado = ahora; return;
  }

  if (fIzq > limiteOponente && fDer > limiteOponente) {
    if (ultimoLado == 1) girarIzquierda(velGiro);
    else girarDerecha(velGiro);
    tiempoUltimoDetectado = ahora; return;
  }

  if (ahora - tiempoUltimoDetectado < persistencia) {
    if (ultimoLado == 1) girarIzquierda(velGiro);
    else girarDerecha(velGiro);
  } else {
    buscarOponente();
  }
  delay(10);
}
