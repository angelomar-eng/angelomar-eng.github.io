03.

Lógica de Control con Bluepad32


// Mapeo de control R2/L2 y D-pad
void loop() {
    BP32.update();
    ControllerPtr myGamepad = myControllers[0];

    if (myGamepad && myGamepad->isConnected()) {
        int r2 = map(myGamepad->throttle(), 0, 740, 0, 1023); // Adelante
        int l2 = map(myGamepad->brake(), 0, 900, 0, 1023);    // Reversa
        int dpad = myGamepad->dpad();

        if (dpad == 4) driveMotor(2, 900);  // Giro Der
        else if (dpad == 8) driveMotor(-2, 900); // Giro Izq
        else if (r2 > 50) driveMotor(1, r2);
        else if (l2 > 50) driveMotor(-1, l2);
        else driveMotor(0, 0);
    }
}

Se implementó un sistema de control basado en Bluetooth HID. La resolución de 10 bits (0-1023) en el PWM permite una aceleración progresiva y precisa.

  • Frecuencia PWM: 20,000 Hz
  • Resolución: 10-bit
  • Driver: Puente H Dual
